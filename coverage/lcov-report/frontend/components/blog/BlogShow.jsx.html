  componentDidMount() {

  fetchBlog() {
    const { props } = this.props;
    let id = props.match.params.id;
    fetch("/blogs/" + id)
      .then(res => {
        return res.json();
      })
      .then(blog => {
        this.setState({ blog });
      })
      .catch(err => this.setState({ error: "There Was an Error..." }));
  }
